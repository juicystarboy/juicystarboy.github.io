89 def get_e_angs(self): #method to read Euler angles from the IMU
90 e_angs_data = self.i2c.mem_read(6, self.address, eul_data_x[1])
91 # combine each pair of bytes into one 16-bit value for x, y, and z
92 e_angs_raw = struct.unpack('<hhh', e_angs_data) #<bc written lsb, msb. /16 to convert to degress (datasheet sec 3.3)

95 roll = e_angs_raw[1]/16
96 pitch = e_angs_raw[2]/16

108 def get_omegas(self): #method to read angular velocity from the IMU
109 omegas_data = self.i2c.mem_read(6, self.address, gyro_data_x[1])
110 omegas_raw = struct.unpack ('<hhh', omegas_data)#/16
